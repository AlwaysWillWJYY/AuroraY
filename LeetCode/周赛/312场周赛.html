    # 一、按身高排序 (opens new window)

    class Solution {
    public String[] sortPeople(String[] names, int[] heights) {
        int n = names.length;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for(int i = 0; i < heights.length; i++) {
            map.put(heights[i], i);
        }
        String[] ans = new String[n];
        int idx = n - 1;
        for(Map.Entry<Integer,Integer> entry : map.entrySet()){
            int key = entry.getKey();
            int value = entry.getValue();
            ans[idx--] = names[value];
        }
        
        return ans;
    }
}

    # 二、按位与最大的最长子数组 (opens new window)

    class Solution {
    public int longestSubarray(int[] nums) {
        int n = nums.length;
        int max = 0, cnt = 0;
        for(int i = 0; i < n; i++) {
            max = Math.max(max, nums[i]);
        }
        int ans = 0;
        for(int i = 0; i <n; i++) {
            if(nums[i] == max) {
                cnt++;
                ans = Math.max(ans, cnt);
            }else{
                cnt = 0;
            }
        }
        return ans;
    }
}

    # 三、找到所有好下标 (opens new window)

    class Solution {
    public List<Integer> goodIndices(int[] nums, int k) {
        int n = nums.length;
        int[] up = new int[n];
        int[] down = new int[n];
        for(int i = 0; i < n; i++) {
            down[i] = 1;
            if(i != 0 && nums[i] <= nums[i - 1]) {
                down[i] = down[i - 1] + 1;
            }
        }
        for(int i = n - 1; i >= 0; i--) {
            up[i] = 1;
            if(i != n - 1 && nums[i] <= nums[i + 1]) {
                up[i] = up[i + 1] + 1;
            }
        }
       
        List<Integer> res = new ArrayList<>();
        for (int i = k; i < n - k; i++) {
            if (down[i - 1] >= k && up[i + 1] >= k)
                res.add(i);
        }
        return res;
    }
}

    # 四、好路径的数目 (opens new window)

    class Solution {
    int[] p;
    public int find(int x) {
        if(p[x] != x) p[x] = find(p[x]);
        return p[x];
    }
    public int numberOfGoodPaths(int[] vals, int[][] edges) {
        //类似于克鲁斯卡尔算法
        int n = vals.length;
        p = new int[n];
        int[][] q = new int[n][2];
        for(int i = 0; i < n; i++) {
            p[i] = i;
            q[i] = new int[]{vals[i], i};
        }
        Arrays.sort(q, (a, b) -> a[0] - b[0]);
        List<Integer>[] li = new List[n];
        for(int i = 0; i < n; i++) {
            li[i] = new ArrayList<>();
        }
        for(int[] g : edges) {
            li[g[0]].add(g[1]);
            li[g[1]].add(g[0]);
        }
        int res = 0;
        for(int i = 0; i < n; i++) {
            int j = i + 1;
            while(j < n && vals[q[i][1]] == vals[q[j][1]]) j++;
            for(int k = i; k < j; k++) {
                //从最大值得每一个点出发
                int x = q[k][1];
                for(int y : li[x]) {
                    if(vals[y] <= vals[x]) {
                        p[find(y)] = find(x);
                    }
                }
            }
            Map<Integer, Integer> map = new HashMap<>();
            for(int k = i; k < j; k++) {
                System.out.println(find(q[k][1]));
                map.put(find(q[k][1]), map.getOrDefault(find(q[k][1]), 0) + 1);
            }
            for(Map.Entry<Integer, Integer> m : map.entrySet()) {
                int v = m.getValue();
                res += v * (v + 1) / 2;
            }
            i = j - 1;
        }
        return res;
    }
}
